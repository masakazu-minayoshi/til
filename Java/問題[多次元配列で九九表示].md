```

import java.util.Scanner;


public class MultidimensionalArray3 {
    public static void main(String[] args) {

    // 配列の宣言。
        int kuku[][] = new int[9][9];

        // 九九の結果を配列に保存。
        for( int i=0; i<9; i++ ) {
            for( int j=0; j<9; j++ ){
                kuku[i][j] = (i + 1) * (j + 1);
                
            }
        }

        // 九九の結果を表示。
        for( int i=0; i<9; i++ ) {
            for( int j=0; j<9; j++ ) {
                System.out.printf("%3d", kuku[i][j]); 
            }
            System.out.println();
        }
	  }
}
```
